```swift ContentView.swift
import Helium

        VStack {

```

```swift App.swift
import SwiftUI
import Helium

@main
struct ExampleApp: App {
    init() {
        let fallbackConfig = HeliumFallbackConfig(
            fallbackPaywall: Text("Loading paywalls...")
        )
        
        Helium.shared.initialize(
            apiKey: "your-api-key",
            heliumPaywallDelegate: StoreKitDelegate(),
            fallbackConfig: fallbackConfig
        )
    }
    
    var body: some Scene {
        WindowGroup {
            ContentView()
        }
    }
}
```
</CodeGroup>

Simple Flutter app demonstrating Helium SDK setup.

<Card title="Flutter Example" icon="flutter" href="https://github.com/cloudcaptainai/helium-flutter/tree/main/example">

<CodeGroup>
```dart main.dart
import 'package:flutter/material.dart';
import 'package:helium_flutter/helium_flutter.dart';

void main() {
  runApp(MyApp());
}

class MyApp extends StatefulWidget {
  @override
  _MyAppState createState() => _MyAppState();
}

class _MyAppState extends State<MyApp> {
  @override
  void initState() {
    super.initState();
    initializeHelium();
  }

  Future<void> initializeHelium() async {
    await HeliumFlutter.initialize(
      apiKey: 'your-api-key',
      fallbackWidget: Container(
        child: Text('Loading paywalls...'),
      ),
    );
  }

  void showPaywall() {
    HeliumFlutter.presentPaywall('main_upgrade');
  }

  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      home: Scaffold(
        body: Center(
          child: ElevatedButton(
            onPressed: showPaywall,
            child: Text('Show Premium Features'),
          ),
        ),
      ),
    );
  }
}
```
</CodeGroup>

## Common Implementation Patterns

### Onboarding Flow

Example of integrating Helium paywalls into an onboarding flow:

```swift
class OnboardingCoordinator {
    func showPaywallAfterOnboarding() {
        // Check if user completed onboarding
        if UserDefaults.standard.bool(forKey: "onboarding_completed") {
            // Show paywall with onboarding-specific trigger
            Helium.shared.presentUpsell(trigger: "post_onboarding")
        }
    }
}
```

### Feature Gating

Pattern for gating features behind paywalls:

```swift
class FeatureManager {
    func accessPremiumFeature() {
        if isPremiumUser() {
            // Allow access
            showPremiumContent()
        } else {
            // Show paywall
            Helium.shared.presentUpsell(trigger: "feature_gate")
        }
    }
}
```

### Trial Expiration

Handling trial expiration with paywalls:

```swift
class TrialManager {
    func checkTrialStatus() {
        let trialEndDate = UserDefaults.standard.object(forKey: "trial_end") as? Date
        
        if let endDate = trialEndDate, Date() > endDate {
            // Trial expired, show paywall
            Helium.shared.presentUpsell(trigger: "trial_expired")
        }
    }
}
```

## Testing Your Integration

### StoreKit Testing

For iOS apps, use StoreKit Configuration files for testing:

1. Create a StoreKit Configuration file in Xcode
2. Add your test products
3. Run your app with the configuration
4. Test purchase flows without charging real money

See our [StoreKit Testing Guide](/sdk/storekit-testing) for detailed instructions.

### Debug Mode

Enable debug logging to troubleshoot integration issues:

```swift
// Enable debug mode
Helium.shared.enableDebugLogging = true
```
